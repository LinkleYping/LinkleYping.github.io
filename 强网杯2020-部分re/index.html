def padding(s,n):  
    len1 = len(s)  
    n1 = len1 % n  
    s += chr(0x1a)*(49-n1)  
    s += chr(0)*(0x40-len(s))

def de_aes_cbc(s,k):  
    if len(s) % 16 != 0:  
        s = padding(s,16)  
    key = k  
    iv = " !\"#$%&'()*+,-./"  
    ci = AES.new(key,AES.MODE_CBC,iv)  
    return ci.decrypt(s)

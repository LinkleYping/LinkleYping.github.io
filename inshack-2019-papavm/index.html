#!/usr/bin/python    
from z3 import *    
    
def work(size):    
  b = [BitVec('%d' % i, 8) for i in xrange(size)]    
  s = Solver()    
  t = BitVecVal(0, 64)    
  p = BitVecVal(1, 32)    
  s.add(b[-1] == 10)  #换行符，因为用的fgets读取    
  for i in xrange(size):    
      p *= 31    
      t += SignExt(32, p)*SignExt(56, b[i])    
  s.add(t == 0xffffffc8b0eb3225)    
  if s.check() == sat:    
      flag = ""    
      ans = s.model()    
      for i in b:    
          flag += chr(ans[i].as_long())    
      print "[*] Found:", flag.encode("base64")    
    
i = 1    
while i < 45:    
  work(i)    
  i += 1
